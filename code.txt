using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;
using System.Windows.Forms;

public partial class ASP期末作業_冷飲店訂單系統Web版_訂單結帳_ : System.Web.UI.Page
{
    protected void Page_Load(object sender, EventArgs e)
    {
        lblLoginName.Text = Session["ID"].ToString();
        if (Session["ProName"] == null)
            Response.Write(" *** 訂單中已無資料！");

        List<CBeverageCartItem> list = (List<CBeverageCartItem>)Session["ProName"];
        GridView1.DataSource = list;
        GridView1.DataBind();
        Response.Write(list);
    }
    protected void btnOrdersRemove_Click(object sender, EventArgs e)
    {
        //List<CBeverageCartItem> list = Session["ProName"] as List<CBeverageCartItem>;
        Session.Clear();
        Session["ID"] = lblLoginName.Text;
        
        Response.Redirect("ASP期末作業_冷飲店訂單系統Web版(訂單查看).aspx");
    }
    protected void lbtnGoBack_Click(object sender, EventArgs e)
    {
        Response.Redirect("ASP期末作業_冷飲店訂單系統Web版(選購).aspx");
    }
    protected void lbtnLogout_Click(object sender, EventArgs e)
    {
        Response.Redirect("ASP期末作業_冷飲店訂單系統Web版(登入).aspx");
    }
    protected void Button1_Click(object sender, EventArgs e)
    {
        
    }
}
